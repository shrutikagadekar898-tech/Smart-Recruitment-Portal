package com.recruitment.ui;

import javax.swing.*;
import javax.swing.border.EmptyBorder;
import javax.swing.table.DefaultTableModel;
import java.awt.*;
import java.sql.*;

public class SmartRecruitmentPortal extends JFrame {

    // ================= DATABASE =================
    private static final String DB_URL = "jdbc:postgresql://localhost:5432/recruitment_portal";
    private static final String DB_USER = "postgres";
    private static final String DB_PASS = "project";

    Connection con;
    CardLayout card = new CardLayout();
    JPanel root = new JPanel(card);

    int loggedUserId = -1;

    // ================= COLORS =================
    Color BG = new Color(245, 247, 252);
    Color HEADER = new Color(40, 53, 147);
    Color BTN = new Color(63, 81, 181);
    Color BTN_ACCENT = new Color(255, 152, 0);

    Font BTN_FONT = new Font("Segoe UI", Font.BOLD, 15);

    public SmartRecruitmentPortal() {
        connectDB();

        setTitle("⭐ STAR SMART RECRUITMENT PORTAL");
        setSize(1250, 780);
        setLocationRelativeTo(null);
        setDefaultCloseOperation(EXIT_ON_CLOSE);

        root.add(startPage(), "START");
        root.add(userLoginPage(), "USER_LOGIN");
        root.add(userRegisterPage(), "USER_REGISTER");
        root.add(candidateDashboard(), "CANDIDATE");
        root.add(jobListPage(), "JOBS");
        root.add(candidateOfferPage(), "USER_OFFERS");

        root.add(companyDashboard(), "COMPANY");
        root.add(postJobPage(), "POST_JOB");
        root.add(interviewPage(), "INTERVIEW");
        root.add(companyOfferPage(), "COMPANY_OFFER");

        root.add(adminLoginPage(), "ADMIN_LOGIN");
        root.add(adminDashboard(), "ADMIN");
        root.add(adminUsersPage(), "ADMIN_USERS");
        root.add(adminCompaniesPage(), "ADMIN_COMPANIES");

        add(root);
        card.show(root, "START");
        setVisible(true);
    }

    void connectDB() {
        try {
            con = DriverManager.getConnection(DB_URL, DB_USER, DB_PASS);
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, "Database connection failed");
        }
    }

    JButton button(String text, Color color) {
        JButton b = new JButton(text);
        b.setBackground(color);
        b.setForeground(Color.WHITE);
        b.setFont(BTN_FONT);
        b.setFocusPainted(false);
        b.setPreferredSize(new Dimension(200, 45));
        return b;
    }

    JPanel header(String title, String backTo) {
        JPanel h = new JPanel(new BorderLayout());
        h.setBackground(HEADER);
        h.setBorder(new EmptyBorder(12, 20, 12, 20));

        JLabel t = new JLabel("⭐ STAR SMART RECRUITMENT PORTAL – " + title);
        t.setForeground(Color.WHITE);
        t.setFont(new Font("Segoe UI", Font.BOLD, 20));

        JButton back = button("Logout", new Color(211, 47, 47));
        back.addActionListener(e -> card.show(root, backTo));

        h.add(t, BorderLayout.WEST);
        h.add(back, BorderLayout.EAST);
        return h;
    }

    JPanel screen(String title, Component center, String backTo) {
        JPanel p = new JPanel(new BorderLayout());
        p.setBackground(BG);
        p.add(header(title, backTo), BorderLayout.NORTH);
        p.add(center, BorderLayout.CENTER);
        return p;
    }

    // ================= START =================

        JPanel c = new JPanel(new GridLayout(4, 1, 20, 20));
        c.setBorder(new EmptyBorder(160, 420, 160, 420));
        c.setBackground(BG);

        t.setFont(new Font("Segoe UI", Font.BOLD, 26));

        JButton u = button("Candidate", BTN);
        JButton co = button("Company", BTN);
        JButton ad = button("Admin", BTN);

        u.addActionListener(e -> card.show(root, "USER_LOGIN"));
        co.addActionListener(e -> card.show(root, "COMPANY_LOGIN"));
        ad.addActionListener(e -> card.show(root, "ADMIN_LOGIN"));

        c.add(t); c.add(u); c.add(co); c.add(ad);

    // ================= USER LOGIN =================
    JPanel userLoginPage() {
        JPanel f = new JPanel(new GridLayout(3, 2, 20, 20));
        f.setBorder(new EmptyBorder(200, 420, 200, 420));

        JTextField email = new JTextField();
        JPasswordField pass = new JPasswordField();

        JButton login = button("Login", BTN_ACCENT);
        JButton reg = button("Register", BTN);

        login.addActionListener(e -> {
            try {
                PreparedStatement ps = con.prepareStatement(
                        "SELECT userid FROM users WHERE email=? AND password=?");
                ps.setString(1, email.getText());
                ps.setString(2, new String(pass.getPassword()));
                ResultSet rs = ps.executeQuery();
                if (rs.next()) {
                    loggedUserId = rs.getInt(1);
                    card.show(root, "CANDIDATE");
                } else JOptionPane.showMessageDialog(this, "Invalid credentials");
            } catch (Exception ex) {
                JOptionPane.showMessageDialog(this, "Login error");
            }
        });

        reg.addActionListener(e -> card.show(root, "USER_REGISTER"));

        f.add(new JLabel("Email")); f.add(email);
        f.add(new JLabel("Password")); f.add(pass);
        f.add(reg); f.add(login);

        return screen("Candidate Login", f, "START");
    }

    JPanel userRegisterPage() {
        JPanel f = new JPanel(new GridLayout(4, 2, 20, 20));
        f.setBorder(new EmptyBorder(180, 420, 180, 420));

        JTextField name = new JTextField();
        JTextField email = new JTextField();
        JPasswordField pass = new JPasswordField();

        JButton b = button("Register", BTN_ACCENT);
        b.addActionListener(e -> {
            try {
                PreparedStatement ps = con.prepareStatement(
                        "INSERT INTO users(name,email,password) VALUES (?,?,?)");
                ps.setString(1, name.getText());
                ps.setString(2, email.getText());
                ps.setString(3, new String(pass.getPassword()));
                ps.executeUpdate();
                JOptionPane.showMessageDialog(this, "Registered Successfully");
                card.show(root, "USER_LOGIN");
            } catch (Exception ex) {
                JOptionPane.showMessageDialog(this, "Email already exists");
            }
        });

        f.add(new JLabel("Name")); f.add(name);
        f.add(new JLabel("Email")); f.add(email);
        f.add(new JLabel("Password")); f.add(pass);
        f.add(new JLabel()); f.add(b);

        return screen("Candidate Register", f, "START");
    }

    // ================= USER DASHBOARD =================
    JPanel candidateDashboard() {
        JPanel g = new JPanel(new GridLayout(1, 2, 40, 40));
        g.setBorder(new EmptyBorder(180, 280, 180, 280));
        g.setBackground(BG);

        JButton jobs = button("View & Apply Jobs", BTN);
        JButton offers = button("My Job Offers", BTN_ACCENT);

        jobs.addActionListener(e -> card.show(root, "JOBS"));
        offers.addActionListener(e -> card.show(root, "USER_OFFERS"));

        g.add(jobs); g.add(offers);
        return screen("Candidate Dashboard", g, "START");
    }

    JPanel jobListPage() {
        DefaultTableModel m = new DefaultTableModel(
                new String[]{"Job ID", "Title", "Skill"}, 0);
        JTable t = new JTable(m);

        try {
            ResultSet rs = con.createStatement()
                    .executeQuery("SELECT jobid,title,skill FROM job");
            while (rs.next())
                m.addRow(new Object[]{rs.getInt(1), rs.getString(2), rs.getString(3)});
        } catch (Exception ignored) {}

        JButton apply = button("Apply Selected Job", BTN_ACCENT);
        apply.addActionListener(e -> {
            int r = t.getSelectedRow();
            if (r == -1) return;
            try {
                PreparedStatement ps = con.prepareStatement(
                        "INSERT INTO application(userid,jobid,status) VALUES (?,?,?)");
                ps.setInt(1, loggedUserId);
                ps.setInt(2, (int) t.getValueAt(r, 0));
                ps.setString(3, "Applied");
                ps.executeUpdate();
                JOptionPane.showMessageDialog(this, "Application Submitted");
            } catch (Exception ex) {
                JOptionPane.showMessageDialog(this, "Already Applied");
            }
        });

        JPanel p = new JPanel(new BorderLayout());
        p.add(new JScrollPane(t), BorderLayout.CENTER);
        p.add(apply, BorderLayout.SOUTH);

        return screen("Job Listings", p, "CANDIDATE");
    }

    JPanel candidateOfferPage() {
        DefaultTableModel m = new DefaultTableModel(
                new String[]{"Offer ID", "Salary", "Joining Date", "Status"}, 0);
        JTable t = new JTable(m);

        try {
            ResultSet rs = con.createStatement()
                    .executeQuery("SELECT offerid,salary,joining_date,offer_status FROM job_offer");
            while (rs.next())
                m.addRow(new Object[]{rs.getInt(1), rs.getDouble(2),
                        rs.getDate(3), rs.getString(4)});
        } catch (Exception ignored) {}

        JButton acc = button("Accept", new Color(56, 142, 60));
        JButton rej = button("Reject", new Color(211, 47, 47));

        acc.addActionListener(e -> updateOffer(t, "Accepted"));
        rej.addActionListener(e -> updateOffer(t, "Rejected"));

        JPanel b = new JPanel();
        b.add(acc); b.add(rej);

        JPanel p = new JPanel(new BorderLayout());
        p.add(new JScrollPane(t), BorderLayout.CENTER);
        p.add(b, BorderLayout.SOUTH);

        return screen("My Job Offers", p, "CANDIDATE");
    }

    void updateOffer(JTable t, String status) {
        int r = t.getSelectedRow();
        if (r == -1) return;
        try {
            PreparedStatement ps = con.prepareStatement(
                    "UPDATE job_offer SET offer_status=? WHERE offerid=?");
            ps.setString(1, status);
            ps.setInt(2, (int) t.getValueAt(r, 0));
            ps.executeUpdate();
            JOptionPane.showMessageDialog(this, "Offer " + status);
        } catch (Exception ignored) {}
    }
